Deklarace funkcí

V TypeScriptu lze funkce deklarovat několika způsoby, podobně jako v JavaScriptu.

// Klasická funkce
function pozdrav(jmeno: string): string {
    return `Ahoj, ${jmeno}!`;
}

// Arrow funkce
const scitani = (a: number, b: number): number => a + b;

// Funkce jako proměnná
const nasobeni: (a: number, b: number) => number = function(a, b) {
    return a * b;
};

Parametry a návratové hodnoty

TypeScript umožňuje specifikovat volitelné parametry a výchozí hodnoty.

// Volitelný parametr s otazníkem
function pozvanka(jmeno: string, cas?: string): string {
    if (cas) {
        return `Ahoj ${jmeno}, zveme tě na akci v ${cas}.`;
    } else {
        return `Ahoj ${jmeno}, zveme tě na naši akci.`;
    }
}

// Výchozí hodnota parametru
function vytvorUzivatele(jmeno: string, vek: number = 18): object {
    return { jmeno, vek };
}

console.log(pozvanka("Jan")); // "Ahoj Jan, zveme tě na naši akci."
console.log(pozvanka("Jana", "15:00")); // "Ahoj Jana, zveme tě na akci v 15:00."
console.log(vytvorUzivatele("Petr")); // { jmeno: "Petr", vek: 18 }
console.log(vytvorUzivatele("Marie", 25)); // { jmeno: "Marie", vek: 25 }

Typy parametrů a návratových hodnot

TypeScript umožňuje přesně specifikovat typy parametrů i návratových hodnot funkcí.

// Základní typování funkcí
function secti(a: number, b: number): number {
    return a + b;
}

// Funkce bez návratové hodnoty
function vypisInformace(zprava: string): void {
    console.log(`INFO: ${zprava}`);
}

// Funkce s více možnými návratovými typy
function prevedHodnotu(hodnota: string | number): string | number {
    if (typeof hodnota === "string") {
        return parseInt(hodnota) || 0;
    } else {
        return hodnota.toString();
    }
}

console.log(secti(5, 3)); // 8
vypisInformace("Operace dokončena"); // INFO: Operace dokončena
console.log(prevedHodnotu("123")); // 123 (jako číslo)
console.log(prevedHodnotu(456)); // "456" (jako řetězec)
